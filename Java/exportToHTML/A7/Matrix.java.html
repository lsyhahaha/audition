11       public Matrix(int n, int m) {
12           System.out.println("调用Matrix类的有参构造方法");
13           this.nums = new int[n][m];
14   
15           for (int i = 0; i < n; i++){
16               for (int j = 0; j < n; j++){
17                   // 获取随机数
18                   double d = Math.random();
19                   int a = (int)(d*100);
20                   nums[i][j] = a;
21               }
22           }
23       }
24   
25       //输出Matrix类中数组的元素值
26       public void output(){
27           int m = this.nums.length;
28           int n = this.nums[0].length;
29   
30           for (int i = 0; i < m; i++){
31               for (int j = 0; j < n; j++){
32                   System.out.print(nums[i][j] + " ");
33               }
34               System.out.print('\n');
35           }
36   
37       }
38   
39       //输出一个矩阵的转置矩阵
40       public void transpose(){
41           int m = this.nums.length;
42           int n = this.nums[0].length;
43   
44           for (int i = 0; i < n; i++){
45               for (int j = 0; j < m; j++){
46                   System.out.print(nums[j][i] + " ");
47               }
48               System.out.print('\n');
49           }
50       }
51   
52       public static void main(String[] args) {
